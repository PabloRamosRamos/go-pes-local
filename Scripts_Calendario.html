// ── PARSEO DE DESCRIPCION DEL EVENTO ───────────────────────────────────────── function parseCalDescripcion_(text) { var result = {}; String(text || '').split(/\r?\n/).forEach(function(line) { var sep = line.indexOf(':'); if (sep < 1) return; var key = line.substring(0, sep).trim().toUpperCase().replace(/\s+/g, '_'); var val = line.substring(sep + 1).trim(); if (key && val) result[key] = val; }); return result; } function buildPrefillFromCalEvent_(ev) { var d = parseCalDescripcion_(ev.descripcion || ''); return { nombre_vecino: d['NOMBRE'] || '', apellido_vecino: d['APELLIDO'] || '', rut_vecino: d['RUT'] || '', telefono_contacto: d['TELEFONO'] || d['TEL'] || '', correo_contacto: d['CORREO'] || d['EMAIL'] || '', uv: d['UV'] || '', tipo_vivienda: d['TIPO_VIVIENDA'] || d['VIVIENDA'] || '', observaciones_form: d['OBSERVACION'] || d['OBS'] || '', direccion_original: ev.lugar || '', fecha_solicitud: ev.fecha || '' }; } // ── DIALOGO DE ASISTENCIA ───────────────────────────────────────────────────── var _calPendingEvent = null; function _calShowDialog(ev) { _calPendingEvent = ev; var dialog = document.getElementById('inicio-cal-dialog'); if (!dialog) return; var titulo = dialog.querySelector('#cal-dialog-titulo'); if (titulo) titulo.textContent = ev.titulo; dialog.querySelector('#cal-step-1').classList.remove('is-hidden'); dialog.querySelector('#cal-step-2').classList.add('is-hidden'); dialog.querySelector('#cal-step-loading').classList.add('is-hidden'); var inp = dialog.querySelector('#cal-asistentes'); if (inp) inp.value = ''; dialog.classList.remove('is-hidden'); var btnSi = dialog.querySelector('#cal-btn-si'); if (btnSi) btnSi.focus(); } function _calHideDialog() { var dialog = document.getElementById('inicio-cal-dialog'); if (dialog) dialog.classList.add('is-hidden'); _calPendingEvent = null; } function _calShowLoading() { var dialog = document.getElementById('inicio-cal-dialog'); if (!dialog) return; dialog.querySelector('#cal-step-1').classList.add('is-hidden'); dialog.querySelector('#cal-step-2').classList.add('is-hidden'); dialog.querySelector('#cal-step-loading').classList.remove('is-hidden'); } function _calRegistrar(realizado, asistentes) { var ev = _calPendingEvent; if (!ev) return; _calShowLoading(); api('registrarEventoCalendario', { eventId: ev.id, realizado: realizado, asistentes: asistentes, titulo: ev.titulo, fecha: ev.fecha }).then(function() { // Quitar el evento del estado local if (APP.state.calendarioData && Array.isArray(APP.state.calendarioData.eventos)) { APP.state.calendarioData.eventos = APP.state.calendarioData.eventos.filter(function(e) { return e.id !== ev.id; }); } _calHideDialog(); // Re-renderizar la sección renderCalendarioRoot_(); // Si se realizó, continuar al form de nuevo ingreso if (realizado) { APP.state.nuevoIngresoPrefill = buildPrefillFromCalEvent_(ev); route('nuevo-ingreso'); } }).catch(function(err) { _calHideDialog(); showError && showError(err); }); } // ── RENDER Y WIRING ─────────────────────────────────────────────────────────── function renderCalendarioRoot_() { var root = document.getElementById('inicio-calendario-root'); if (!root || !window.GO_PES_CALENDARIO) return; function onCalAction(eventId) { var eventos = (APP.state.calendarioData && APP.state.calendarioData.eventos) || []; var ev = null; for (var i = 0; i < eventos.length; i++) { if (eventos[i].id === eventId) { ev = eventos[i]; break; } } if (!ev) return; _calShowDialog(ev); } GO_PES_CALENDARIO.render(root, APP.state.calendarioData, escapeHtml, onCalAction); // Wiring del dialogo (una sola vez por root) if (!root.dataset.calDialogWired) { root.dataset.calDialogWired = '1'; root.addEventListener('click', function(e) { // Paso 1: Sí if (e.target.closest('#cal-btn-si')) { var dialog = document.getElementById('inicio-cal-dialog'); dialog.querySelector('#cal-step-1').classList.add('is-hidden'); dialog.querySelector('#cal-step-2').classList.remove('is-hidden'); var inp = dialog.querySelector('#cal-asistentes'); if (inp) { inp.value = ''; inp.focus(); } return; } // Paso 1: No if (e.target.closest('#cal-btn-no')) { _calRegistrar(false, 0); return; } // Paso 2: Continuar if (e.target.closest('#cal-btn-continuar')) { var inp2 = document.getElementById('cal-asistentes'); var n = Math.max(0, parseInt(inp2 ? inp2.value : '0', 10) || 0); _calRegistrar(true, n); return; } // Cancelar (ambos pasos) if (e.target.closest('#cal-btn-cancel-1') || e.target.closest('#cal-btn-cancel-2')) { _calHideDialog(); return; } }); // Cerrar con Escape document.addEventListener('keydown', function(e) { if (e.key === 'Escape') _calHideDialog(); }); } if (APP.state.calendarioData !== null) return; api('getCalendarioEventos', {}) .then(function(data) { APP.state.calendarioData = data || {}; APP.state.calendarioFetchedAt = Date.now(); GO_PES_CALENDARIO.render(root, APP.state.calendarioData, escapeHtml, onCalAction); }) .catch(function() { APP.state.calendarioData = { eventos: [] }; APP.state.calendarioFetchedAt = Date.now(); GO_PES_CALENDARIO.render(root, APP.state.calendarioData, escapeHtml, onCalAction); }); }
